dbo.ActualizarMasivoGPSDataFlag

  • Collapse All
  • Expand All

Description

Procedimiento almacenado que una vez enviada la información de los puntos GPS al DIG Server, actualiza el estado de registro enviado a Geotab en la Tabla GPS

Parameters

Name Data Type Length Description
@DriverTripID int 4 Identificación del viaje
@GPSDataTime varchar 23 Fecha en formato UTC del dato GPS enviado exitosamente al DIG Server

SET QUOTED_IDENTIFIER, ANSI_NULLS ON

CREATE OR ALTER PROCEDURE dbo.ActualizarMasivoGPSDataFlag
( @DriverTripID INT,
    @GPSDataTime VARCHAR(23)
)
AS
BEGIN
    -- Imprimir los valores de los parámetros
    PRINT 'DriverTripID: ' + CONVERT(VARCHAR(10), @DriverTripID)
    PRINT 'GPSDataTime: ' + @GPSDataTime

    -- Declarar variables locales
    DECLARE @Speed INT;
    DECLARE @HDOP INT;
    
    -- Obtener valores de configuración
    SELECT @Speed = Speed, @HDOP = HDOP FROM dbo.Settings;

    -- Verificar si existe un registro en dbo.GPSData con los parámetros especificados
   /* IF EXISTS ( SELECT 1 FROM dbo.GPSData WHERE DriverTripID = @DriverTripID AND GPSDataTime = @GPSDataTime )*/
    BEGIN
        -- Declarar variable para almacenar el valor de Flag
        DECLARE @Flag INT;

        -- Obtener el valor de Flag
        SELECT TOP 1 @Flag = Flag
        FROM dbo.GPSData
        WHERE DriverTripID = @DriverTripID
        AND GPSDataTime = @GPSDataTime
        AND isOnGeoTabServer = 0;

        -- Actualizar el registro en función del valor de Flag
        UPDATE dbo.GPSData
        SET isOnGeoTabServer = 1
        WHERE DriverTripID = @DriverTripID
        AND GPSDataTime = @GPSDataTime
        AND Flag = @Flag
        AND (
            (@Flag = 0)
            OR 
            (@Flag = 1 AND Speed > @Speed AND HDOP < @HDOP)
            OR
            (@Flag = 2 AND isIgnitionOn = 0)
        );
    END;
END;

--EXEC ActualizarMasivoGPSDataFlag 63, ' 2024-03-01 20:19:06.000'
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Procedimiento almacenado que una vez enviada la información de los puntos GPS al DIG Server, actualiza el estado de registro enviado a Geotab en la Tabla GPS', 'SCHEMA', N'dbo', 'PROCEDURE', N'ActualizarMasivoGPSDataFlag'
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Identificación del viaje', 'SCHEMA', N'dbo', 'PROCEDURE', N'ActualizarMasivoGPSDataFlag', 'PARAMETER', N'@DriverTripID'
GO

EXEC sys.sp_addextendedproperty N'MS_Description', N'Fecha en formato UTC del dato GPS enviado exitosamente al DIG Server', 'SCHEMA', N'dbo', 'PROCEDURE', N'ActualizarMasivoGPSDataFlag', 'PARAMETER', N'@GPSDataTime'
GO
